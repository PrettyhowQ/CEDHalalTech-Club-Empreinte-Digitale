  <script>
    const traductions = {
      fr: "Bienvenue chez Costa del Sol",
      en: "Welcome to Costa del Sol",
      ar: "مرحبا بكم في كوستا ديل سول",
      es: "Bienvenido a Costa del Sol",
      zh: "欢迎来到阳光海岸"
    };
    function changerLangue() {
      const lang = document.getElementById("lang-select").value;
      const texte = traductions[lang] || traductions['fr'];
      document.getElementById("texte-bienvenue").textContent = texte;
    }
  </script>

  <script>
    document.addEventListener("DOMContentLoaded", function () {
      const langSelect = document.getElementById("lang-select");
      const savedLang = localStorage.getItem("selectedLang");
      if (savedLang) {
        langSelect.value = savedLang;
        changerLangue();
      }
      langSelect.addEventListener("change", function () {
        localStorage.setItem("selectedLang", this.value);
        changerLangue();
      });
    });
  </script>
